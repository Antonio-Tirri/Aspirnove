37 /* Set extra character space pixel columns (default = 0) */
38 void gsetspch(GXT chsp)
39  {
40  if (gcurvp == NULL)
41  return;
42  #ifdef GDATACHECK
43  if (chsp >= gcurvp->rb.x - gcurvp->lt.x)
44  {
45  G_WARNING( "gsetspch: character spacing exceeds viewport size" );
46  chsp = gcurvp->rb.y - gcurvp->lt.y;
47  }
48  #endif
49  gcurvp->chln.x = chsp;
50  gi_calcdatacheck(); /* correct VP to new settings */
51  }
