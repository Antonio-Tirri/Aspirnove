25  Set character pos. in view-port in a pixel based fashion.
26  Calculations and range limits are done relative to the current viewport
27 */
28 void gsetpos( GXT xpos, GYT ypos )
29  {
30  gi_datacheck(); /* check internal data for errors */
31  /* normalize to view-port */
32  xpos = xpos + gcurvp->lt.x;
33  ypos = ypos + gcurvp->lt.y;
34 
35  #if defined( GBASIC_TEXT ) && !defined( GHW_NO_HDW_FONT )
36  if( gishwfont() )/* adjust to correct pos */
37  {
38  ghw_setcabspos( xpos, ypos );
39  xpos = ghw_getcursorxpos(); /* back to abs pixel pos */
40  ypos = ghw_getcursorypos();
41  }
42  #endif
43  if( xpos < gcurvp->lt.x )
44  {
45  G_WARNING( "gsetpos: parameter, x<vp.left" );
46  xpos = gcurvp->lt.x;
47  }
48  if( xpos > gcurvp->rb.x )
49  {
50  G_WARNING( "gsetpos: parameter, x>cp.right" );
51  xpos = gcurvp->rb.x;
52  }
53  if( ypos < gcurvp->lt.y )
54  {
55  G_WARNING( "gsetpos: parameter, y<vp.top" );
56  ypos = gcurvp->lt.y;
57  }
58  if( ypos > gcurvp->rb.y )
59  {
60  G_WARNING( "gsetpos: parameter, y>vp.bottom" );
61  ypos = gcurvp->rb.y;
62  }
63 
64  gi_cursor( 0 ); /* kill cursor */
65 
66  #ifdef GGRAPHICS
67  gcurvp->ppos.x = xpos; /* update graphics pos also */
68  gcurvp->ppos.y = ypos;
69  #endif
70 
71  gcurvp->cpos.x = xpos;
72  #ifndef GNOTXTSPACE
73  if (gcurvp->chln.y != 0)
74  {
75  /* Compensate for extra line space */
76  if (gcurvp->lt.y+gcurvp->chln.y > ypos)
77  gcurvp->cpos.y = gcurvp->lt.y;
78  else
79  gcurvp->cpos.y = ypos - gcurvp->chln.y;
80  }
81  else
82  #endif
83  gcurvp->cpos.y = ypos;
84 
85 
86  gi_calcdatacheck(); /* correct VP to new settings */
87  gi_cursor( 1 ); /* set cursor */
88  }
