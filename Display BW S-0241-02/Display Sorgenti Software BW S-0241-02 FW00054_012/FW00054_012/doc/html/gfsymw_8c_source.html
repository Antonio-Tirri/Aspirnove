42  Finds a pointer to char
43  Must not be called if it is a hardware font
44 */
45 PGSYMBOL gi_getsymbol( GWCHAR c, PGFONT pfont, PGCODEPAGE codepagep)
46  {
47  PGSYMBYTE ps;
48  GBUFINT s;
49  PGCP_RANGE crpb;
50  #ifdef USE_LINEAR_SEARCH
51  GWCHAR cnt;
52  #endif
53  #if !defined( USE_LINEAR_SEARCH ) || defined( GVIRTUAL_FONTS )
54  GWCHAR numelem, min, half;
55  PGCP_RANGE crp;
56  #endif
57 
58 
59  #ifdef GVIRTUAL_FONTS
60  PGCODEPAGEV cpv;
61  PGCP_RANGE_V crpv;
62  if (gisfontv(pfont))
63  {
64  /* Prepare virtual font driver for codepage and symbol lookup */
65  #ifdef GDATACHECK
66  /* Check for mixed font design error */
67  if (((((PGFONTV)(pfont))->type_id) & GVTYPEMASK) != 1)
68  { /* Error: format is unknown to this library */
69  G_WARNING( "Unsupported extended font format. Illegal virtual font" );
70  return NULL;
71  }
72  if ((((PGFONTV)(pfont))->numbits) != 0)
73  { /* Error Color symbols are not supported by b&w libs */
74  G_WARNING( "Color symbols are not supported by b&w libs" );
75  return NULL;
76  }
77  #endif
78  gi_fontv_open((PGFONTV) pfont);
79  if( codepagep != NULL )
80  {
81  if (gisfontcpv(codepagep))
82  {
83  /* Codepage is also in virtual memory */
84  cpv = (PGCODEPAGEV) codepagep;
85  /* Do fast codepage lookup in virtual memory */
86  for(;;)
87  {
88  /* Sub range = full range */
89  numelem = cpv->cprnum;
90  min = 0;
91  for(;;)
92  {
93  half = numelem >> 1; /* half = mid point in sub range */
94  crpv = gi_fontv_cp(min+half); /* Next element to check */
95  if ((c >= crpv->min) && (c <= crpv->max))
96  {
97  /* Found */
98  c = crpv->idx + (c - crpv->min); /* Convert to symbol index */
99  goto c_converted; /* fast skip of all loops */
100  }
101  else
102  {
103  if (half != 0)
104  {
105  /* define next sub range */
106  if (c < crpv->min)
107  numelem = half; /* too high, continue in lower half */
108  else
109  {
110  min = min+half; /* too low, continue in upper half */
111  numelem = numelem - half;
112  }
113 
114  if (numelem != 0) /* subrange exist ? */
115  continue;
116  }
117 
118  /* Not found */
119  if (c == cpv->def_wch)
120  {
121  /* Default character does not exist (either) */
122  G_WARNING("Illegal codepage default character detected");
123  return NULL;
124  }
125  else
126  {
127  /* Use default character instead */
128  c = cpv->def_wch;
129  break; /* Search once more in full range, skip inner loop */
130  }
131  }
132  }
133  }
134  }
135  }
136  }
137  /* We end here if no codepage is used, or if codepage is not in virtual memory */
138  #endif
139 
140  if( codepagep != NULL )
141  {
142  /* Using codepage, convert character to an index via lookup */
143  crpb = &codepagep->cpr[0]; /* Codepage element array base */
144 
145  if (codepagep->cph.cprnum == 0)
146  {
147  G_WARNING("Illegal codepage header detected");
148  return NULL;
149  }
150 
151  #ifdef USE_LINEAR_SEARCH
152  /* Old linear search method */
153  for(;;)
154  {
155  /* Check ranges */
156  for (cnt = 0; cnt < codepagep->cph.cprnum; cnt++)
157  {
158  if ((c >= crpb->min) && (c <= crpb->max))
159  {
160  /* Found */
161  c = crpb->idx + (c - crpb->min); /* found, return symbol index */
162  goto c_converted; /* fast skip of all loops */
163  }
164  crpb++;
165  }
166 
167  /* Not found*/
168  if (c == codepagep->cph.def_wch)
169  {
170  /* Default character does not exist (either) */
171  G_WARNING("Illegal codepage default character detected");
172  return NULL;
173  }
174 
175  /* Use default character instead, and try one more */
176  c = codepagep->cph.def_wch;
177  crpb = &codepagep->cpr[0];
178  }
179 
180  #else
181 
182  /* Use fast lookup
183  Much faster with large codepages but require that codepage elements are
184  arranged in increasing order (= default ordering for all standard font
185  code pages, and for the fonts codepages created with IconEdit) */
186  for(;;)
187  {
188  /* Sub range = full range */
189  numelem = codepagep->cph.cprnum;
190  min = 0;
191  for(;;)
192  {
193  half = numelem >> 1; /* half = mid point in sub range */
194  crp = &crpb[(SGUINT)min+half]; /* Next element to check */
195  if ((c >= crp->min) && (c <= crp->max))
196  {
197  /* Found */
198  c = crp->idx + (c - crp->min); /* Convert to symbol index */
199  goto c_converted; /* fast skip of all loops */
200  }
201  else
202  {
203  if (half != 0)
204  {
205  /* define next sub range */
206  if (c < crp->min)
207  numelem = half; /* too high, continue in lower half */
208  else
209  {
210  min = min+half; /* too low, continue in upper half */
211  numelem = numelem - half;
212  }
213 
214  if (numelem != 0) /* subrange exist ? */
215  continue;
216  }
217 
218  /* Not found */
219  if (c == codepagep->cph.def_wch)
220  {
221  /* Default character does not exist (either) */
222  G_WARNING("Illegal codepage default character detected");
223  return NULL;
224  }
225  else
226  {
227  /* Use default character instead */
228  c = codepagep->cph.def_wch;
229  break; /* Search once more in full range, skip inner loop */
230  }
231  }
232  }
233  }
234  #endif /* USE_LINEAR_SEARCH */
235  }
236  else
237  if ( c > gi_fnumsym(pfont))
238  {
239  G_WARNING( "gfsymv.c: symbol index larger than font table" );
240  return NULL;
241  }
242 
243  c_converted:
244  #ifdef GVIRTUAL_FONTS
245  if (gisfontv(pfont))
246  return gi_fontv_sym(c); /* Do GFONTV lookup */
247  /* Do GFONT lookup */
248  #endif
249 
250  /* Is a standard font (linear symbol array in linear memory) */
251  s = sizeof(GSYMHEAD) /* Sizeof symbol header */
252  + ((GBUFINT) pfont->symsize); /* size of symbol data */
253  ps = (PGSYMBYTE) (pfont->psymbols);
254  return (PGSYMBOL)(&ps[s*c]);
255  }
