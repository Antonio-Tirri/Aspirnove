3  Graphic block copy functions for LCD display
4 
5  Read graphic area from the display to a GLCD buffer.
6  Write graphic buffer to LCD display.
7 
8  Information about the size of the graphic area is stored in the buffer.
9  The buffer can be written back to the display with ghw_wrblk(),
10  optionally with another start origin.
11 
12  All coordinates are absolute pixel coordinate.
13 
14  ---------
15 
16  The KS07XX controller is assumed to be used with a LCD module.
17 
18  The following LCD module characteristics MUST be correctly
19  defined in GDISPCFG.H:
20 
21  GDISPW Display width in pixels
22  GDISPH Display height in pixels
23  GBUFFER If defined most of the functions operates on
24  a memory buffer instead of the LCD hardware.
25  The memory buffer content is complied to the LCD
26  display with ghw_updatehw().
27  (Equal to an implementation of delayed write)

81 /*
82  Copy a graphic area from the display to a GLCD buffer
83  Information about the size of the graphic area is saved in the buffer.
84  The buffer can be written back to the display with ghw_wrblk(),
85  optionally with another start origin.
86 
87  All coordinates are absolute pixel coordinate.
88 
89  The first part of the buffer will be a dynamic header defining
90  the block rectangle:
91  GXT left_top_x,
92  GYT left_top_y,
93  GXT right_bottom_x,
94  GYT right_bottom_y,
95  followed by th block data
96 */
97 void ghw_rdblk(GXT ltx, GYT lty, GXT rbx, GYT rby, SGUCHAR *dest, GBUFINT bufsize )
98  {
99  GYT y;
100  GXT x;
101  PGHW_BLK_HEADER desthdr;
102  #ifdef GBUFFER
103  GBUFINT gbufidx;
104  GBUF_CHECK();
105  #endif
106 
107  glcd_err = 0;
108  if (dest == NULL)
109  return;
110 
111  /* Force reasonable values */
112  GLIMITU(ltx,GDISPW-1);
113  GLIMITU(lty,GDISPH-1);
114  GLIMITD(rby,lty);
115  GLIMITU(rby,GDISPH-1);
116  GLIMITD(rbx,ltx);
117  GLIMITU(rbx,GDISPW-1);
118 
119  if (ghw_blksize(ltx, lty, rbx, rby) > bufsize)
120  {
121  G_ERROR( "ghw_rdblk: dest buffer too small" );
122  return;
123  }
124 
125  /* Save header info */
126  desthdr = (PGHW_BLK_HEADER) dest;
127  dest = &(desthdr->dat[0]);
128  desthdr->lx = ltx;
129  desthdr->ly = lty;
130  desthdr->rx = rbx;
131  desthdr->ry = rby;
132 
133  lty &= ~(0x7);
134  rby &= ~(0x7);
135  for (y = lty; y <= rby; y+=GDISPCH)
136  {
137  #ifdef GBUFFER
138  gbufidx = GINDEX(ltx, y/8);
139  for (x = ltx; x <= rbx; x++)
140  *dest++ = gbuf[gbufidx++];
141 
142  #else
143  ghw_set_xypos(ltx,y);
144  ghw_auto_rd_start(); /* init for read with auto increment */
145  for (x = ltx; x <= rbx; x++)
146  {
147  *dest++ = ghw_auto_rd(); /* read with auto increment */
148  }
149  #endif
150  }
151  *dest = 0; /* tail = 0 (needed for simple wrblk shift) */
152  }

154 /*
155  Copy a graphic area from a GLCD buffer to the display
156  The GLCD buffer must have been read with ghw_rdblk
157 
158  If the destination range is larger than the buffered range
159  then the destination range is limited to fit the size of
160  the buffered range.
161 
162  If the destination range is smaller than the buffered range
163  then only the upper-left part of the buffer is written to
164  the display.
165 
166  All coordinates are absolute pixel coordinate.
167 
168  The first part of the buffer will be a dynamic header defining
169  the block rectangle:
170  GXT left_top_x,
171  GYT left_top_y,
172  GXT right_bottom_x,
173  GYT right_bottom_y,
174  followed by th block data
175 */
176 void ghw_wrblk(GXT ltx, GYT lty, GXT rbx, GYT rby, SGUCHAR *src )
177  {
178  GXT w,we,x,xe;
179  GYT h,he;
180  SGUCHAR msk;
181  SGINT shift;
182  SGUINT y;
183  SGUINT dat;
184  SGUCHAR datb;
185  PGHW_BLK_HEADER srchdr;
186  #ifdef GBUFFER
187  GBUFINT gbufidx;
188  GBUF_CHECK();
189  #endif
190 
191  glcd_err = 0;
192  if (src == NULL)
193  return;
194 
195  /* Force reasonable values */
196  GLIMITU(ltx,GDISPW-1);
197  GLIMITU(lty,GDISPH-1);
198  GLIMITD(rby,lty);
199  GLIMITD(rbx,ltx);
200  GLIMITU(rby,GDISPH-1);
201  GLIMITU(rbx,GDISPW-1);
202 
203  #ifdef GBUFFER
204  invalrect( ltx, lty );
205  invalrect( rbx, rby );
206  #endif
207 
208  /* Get header info about stored buffer */
209  srchdr = (PGHW_BLK_HEADER) src;
210  src = &(srchdr->dat[0]);
211  w = srchdr->lx;
212  h = srchdr->ly;
213  we = srchdr->rx;
214  he = srchdr->ry;
215 
216  /* Set shift value (negative = shift left) */
217  shift = (SGINT)(h % GDISPCH) - (SGINT)(lty % GDISPCH);
218 
219  /* Limit destination range against source window size in buffer */
220  if (rbx-ltx > (we-w))
221  rbx = ltx + (we-w);
222  if (rby-lty > (he-h))
223  rby = lty + (he-h);
224 
225  xe = rbx - ltx;
226  w = we - w + 1; /* Stored line width in bytes */
227 
228  /* Select stop and start mask */
229  msk = startmask[lty & 0x7];
230  he = stopmask[rby & 0x7];
231  rby /= GDISPCH;
232  lty /= GDISPCH;
233 
234  for (y = lty; y <= rby; y++)
235  {
236  if (y == rby)
237  msk &= he; /* Use stop mask */
238 
239  #ifdef GBUFFER
240  gbufidx = GINDEX(ltx,y);
241  #else
242  ghw_set_xypos(ltx,(GYT)(y*GDISPCH));
243  #endif
244 
245  for (x = 0; x <= xe; x++ )
246  {
247  if (shift != 0)
248  {
249  if (shift > 0)
250  {
251  dat = (SGUINT) src[x];
252  dat |= ((SGUINT) src[(SGUINT)x+w])*256;
253  dat >>= shift;
254  datb = (SGUCHAR) dat;
255  }
256  else
257  {
258  dat = (y == lty) ? 0 : (SGUINT) src[x-w];
259  dat |= ((SGUINT) src[x])*256;
260  dat <<= abs(shift);
261  datb = (SGUCHAR) (dat / 256);
262  }
263  }
264  else
265  datb = src[x];
266 
267  /* Fetch byte and mask */
268  #ifdef GBUFFER
269  if (msk != 0xff)
270  datb = (gbuf[gbufidx] & ~msk) | (datb & msk);
271  gbuf[gbufidx++] = datb;
272  #else
273  if (msk != 0xff)
274  datb = (ghw_rd() & ~msk) | (datb & msk);
275  ghw_auto_wr( datb );
276  #endif
277  }
278  src = &src[w]; /* Set to next buffer line */
279  msk = 0xff;
280  }
281  #if (defined(GBUFFER) || defined(GHW_FAST_SIM_UPDATE))
282  ghw_updatehw();
283  #endif
284  }
