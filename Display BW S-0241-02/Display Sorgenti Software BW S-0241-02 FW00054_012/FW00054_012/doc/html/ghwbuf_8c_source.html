3  Low-level driver functions for the KS07XX LCD display buffer
4  handling. Contains low-level functions for optimized buffer
5  handling in buffered mode (GBUFFER defined) when softfonts and
6  graphic is used.
7 
8  NOTE: These functions are only called by the GDISP high-level
9  functions. They should not be used directly from user programs.
10 
11  The KS07XX controller is assumed to be used with a LCD module.
12  The following LCD module characteristics MUST be correctly
13  defined in GDISPCFG.H:
14 
15  GDISPW Display width in pixels
16  GDISPH Display height in pixels
17  GBUFFER If defined most of the functions operates on
18  a memory buffer instead of the LCD hardware.
19  The memory buffer content is copied to the LCD
20  display with ghw_updatehw().
21  (Equal to an implementation of delayed write)

53 void ghw_updatehw(void)
54  {
55  SGUCHAR *adr; /* fast pointer to xdata */
56  GXT xcnt;
57  GYT y;
58 
59  if (ghw_upddelay)
60  return;
61 
62  GBUF_CHECK();
63  glcd_err = 0;
64 
65  if( irby < ilty ) return;
66  if( irbx < iltx ) return;
67  if( irby >= GDISPH ) irby = GDISPH-1;
68  if( irbx >= GDISPW ) irbx = GDISPW-1;
69 
70  /* Mirror buffer */
71  /* Loop byte rows */
72  ilty /= 8;
73  irby /= 8;
74  for (y = ilty; y <= irby; y++)
75  {
76  ghw_set_xypos(iltx,(GYT)(y*8));
77  adr = &gbuf[ GINDEX(iltx,y)];
78 
79  xcnt = irbx - iltx + 1;
80  /* Loop columns*/
81  while( xcnt-- != 0 )
82  ghw_auto_wr(*adr++);
83  }
84 
85  iltx = 1;
86  ilty = 1;
87  irbx = 0;
88  irby = 0;
89  #if (defined(_WIN32) && defined(GHW_PCSIM))
90  GSimFlush();
91  #endif
92  }
93 
94 /*
95  Set updatehw to instant update or delayed update
96  1 = Normal screen update from buffer
97  0 = Update from buffer stopped until normal update is selected again
98 
99  Activated from gsetupdate(on);
100 */
101 GUPDATE ghw_setupdate( GUPDATE update )
102  {
103  GUPDATE old_update;
104  old_update = (GUPDATE) (ghw_upddelay == 0);
105  if ((update != 0) && (ghw_upddelay != 0))
106  {
107  /* Update is re-activated, make a screen update. */
108  ghw_upddelay = 0; /* Flag = 0 -> no delayed update */
109  ghw_updatehw();
110  }
111  else
112  ghw_upddelay = (update == 0) ? 1 : 0;
113  return old_update;
114  }
