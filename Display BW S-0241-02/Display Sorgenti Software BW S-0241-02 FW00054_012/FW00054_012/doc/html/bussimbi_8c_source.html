2 
3  Generic template file for external bus simulator drivers.
4 
5  The functions in this module is called by single-chip-processor version
6  of the ghwinit.c module.
7 
8  This implementation assumes that the processor port architecture supports
9  BIT addressing capabilities and that BIT addressing is also supported by
10  the target compiler.
11 
12  The compiler independent SG syntax notation is used in this example for
13  clarification.
14  (Alternatively these functions could be implemented using the compiler
15  specific syntax for I/O port operations.)
16 
17  The following symbolic names for bit and byte ports must be declared for
18  the actual processor architecture and target system:
19 
20  Bit ports:
21  CD (LCD address line)
22  NWR (LCD /WR line)
23  NRD (LCD /RD line)
24  NCE (LCD /CE line)
25  NRST (LCD /RST line, if connected to a port)
26 
27  Byte ports:
28  DATAWR (LCD data write)
29  DATARD (LCD data read. May be equal to DATAWR, depends on the processor)
30  DATADIR (Processor data port direction setup, depends on the processor).
31  CTRLDIR (Processor ctrl port direction setup, depends on the processor).

54 #define DATABUS_IS_INP 0x00 /* 8 signal lines is used as input */

58 /*
59  Simulate a bus write operation for a LCD controller with an Intel
60  like bus interface (i.e. use of separate /RD and /WR strobes).
61 
62  The address parameter adr is assumed to be either 0 or 1.
63 */
64 void simwrby(SGUCHAR adr, SGUCHAR dat)
65  {
66  /* A: Set C/D line according to adr, Set /CE line active low */
67  sgwrbi(CD, (adr != 0));
68  sgwrbi(NCE, 0);
69  /* B1: Make data port an output (if required by port architecture) */
70  sgwrby(DATADIR,DATABUS_IS_OUTP);
71  /* B2: Write data to data port */
72  sgwrby(DATAWR,dat);
73  /* C: Set /WR active low, (Delay min 80 ns), */
74  sgwrbi(NWR, 0); /* repeat this instruction if more delay is needed*/
75  /* D: Set /WR passive high */
76  sgwrbi(NWR, 1);
77  /* E: Set /CE passive high */
78  sgwrbi(NCE, 1);
79  }
80 
81 /*
82  Simulate a bus read operation for a LCD controller with an Intel
83  like bus interface (i.e. use of separate /RD and /WR strobes).
84 
85  The address parameter adr is assumed to be either 0 or 1.
86 */
87 SGUCHAR simrdby(SGUCHAR adr)
88  {
89  SGUCHAR dat;
90  /* a: Set C/D line according to adr. Set /CE line active low */
91  sgwrbi(CD, (adr != 0));
92  sgwrbi(NCE, 0);
93  /* b: Make data port an input (if required by port architecture) */
94  sgwrby(DATADIR,DATABUS_IS_INP);
95  /* c: Set /RD active low, (Delay min 150ns), */
96  sgwrbi(NRD, 0); /* repeat this instruction if more delay is needed*/
97  /* d: Read data from data port */
98  dat = sgrdby(DATARD);
99  /* e1: Set /RD passive high */
100  sgwrbi(NRD, 1);
101  /*, Set /CE passive high (could be ignored) */
102  sgwrbi(NCE, 1);
103  return dat;
104  }

106 /*
107  Initialize and reset LCD display.
108  Is called before simwrby() and simrdby() is invoked for the first time
109 
110  The SED1335 reset line is toggled here if it connected to a bus port.
111  (it may alternatively be hard-wired to the reset signal to the processors
112  in the target system).
113 
114  The sim_reset() function is invoked automatically via the ginit() function.
115 */
116 void sim_reset( void )
117  {
118  /* 1. Init data port setup (if required by port architecture) */
119  sgwrby(DATADIR,DATABUS_IS_INP);
120  /* 2. Make C/D, /RD, /WR, /CE (+ FS and /RST) to outputs (if required by port architecture) */
121  sgwrby(CTRLDIR,CTRLBUS_IS_OUTP);
122  sgwrbi(NCD, 1);
123  sgwrbi(NWR, 1);
124  sgwrbi(NRD, 1);
125  sgwrbi(NCE, 1);
126  /* 3. Set LCD reset line /RST active low (if /RST is connected to a port bit) */
127  sgwrbi(NRST, 0);
128  /* 4. Set LCD reset line /RST passive high (if /RST is connected to a port bit) */
129  sgwrbi(NRST, 1);
130  }
