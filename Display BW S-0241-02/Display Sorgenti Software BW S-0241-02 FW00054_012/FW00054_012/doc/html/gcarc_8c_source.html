3  Draw corner arc (a 90 degree arc aligned to x,y axis)

5  GLINE Perimeter line with foreground color (default)

35  {
36  r=xc;
37  G_WARNING( "gcarc: radius turncated" );
38  }
39  if (((arctype & (GCARC_LT|GCARC_RT))!= 0) && ( r > yc))
40  {
41  r=yc;
42  G_WARNING( "gcarc: radius turncated" );
43  }
44 
45  xc = xc + gcurvp->lt.x;
46  yc = yc + gcurvp->lt.y;
47 
48  /* Limit center to be within viewport */
49  LIMITTOVP( "gcarc",xc,yc,xc,yc );
50 
51  /* Check and limit radius to prevent viewport overflow */
52  if (((arctype & (GCARC_RT|GCARC_RB))!= 0)&&(xc+r > gcurvp->rb.x))
53  {
54  r = gcurvp->rb.x-xc;
55  G_WARNING( "gcarc: radius turncated" );
56  }
57  if (((arctype & (GCARC_LB|GCARC_RB))!= 0)&& (yc+r > gcurvp->rb.y))
58  {
59  r = gcurvp->rb.y-yc;
60  G_WARNING( "gcarc: radius turncated" );
61  }
62 
63  if( r == 0)
64  return;
65
