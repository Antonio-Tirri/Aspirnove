25 /* Get vp number for viewport where application data is placed */
26 SGUCHAR ggetvpapp( PGVPAPP app )
27  {
28  SGUCHAR i;
29  /* find current vp index */
30  for( i=0; i<GNUMVP; i++ )
31  {
32  if( app == &gdata.viewports[i].vpapp )
33  break;
34  }
35  if( i<GNUMVP )
36  return i; /* Return viewport which app points to */
37  else
38  {
39  /* The warning message here can be commented out if it is legal for
40  the application to use fallback to a default viewport */
41  G_WARNING( "ggetvpapp: parameter, Illegal PGVPAPP" );
42 
43  return 0; /* Use viewport 0 as the default viewport, or to signal lookup failure */
44  }
45  }
