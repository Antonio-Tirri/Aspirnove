31 /*
32  Return width of cursor in current viewport
33 */
34 GXT gcursor_width( void )
35  {
36  GXT w;
37  #ifndef GHW_NO_HDW_FONT
38  if( gishwfont() )
39  w = GDISPCW;
40  else
41  #endif
42  if (ghw_cursor & GCURBLOCK)
43  w = gi_fsymw(gcurvp->pfont)-1;/* cursor width = font width */
44  else
45  if (ghw_cursor & GCURVERTICAL)
46  w = (GXT)(ghw_cursor&0x0F); /* cursor width = setting */
47  else
48  {
49  w = gi_fsymw(gcurvp->pfont); /* cursor width = font width */
50  if (w > 5) w -= 2; /* Adjust look */
51  }
52  return w;
53  }

55 /*
56  Return height of cursor in current viewport
57 */
58 GYT gcursor_height( void )
59  {
60  GYT h;
61  #ifndef GHW_NO_HDW_FONT
62  if( gishwfont() )
63  h = GDISPCH;
64  else
65  #endif
66  if (ghw_cursor & GCURBLOCK)
67  h = gi_fsymh(gcurvp->pfont)-1; /* cursor height = font height */
68  else
69  if (ghw_cursor & GCURVERTICAL)
70  {
71  h = gi_fsymh(gcurvp->pfont); /* cursor height = font height */
72  if (h > 6) h -= 2; /* Adjust look */
73  }
74  else
75  h = (GYT)((ghw_cursor&0x0F)+1); /* cursor height = setting */
76  return h;
77  }

79 /********************************************************************
80 internal function
81 */
82 void gi_cursor( SGBOOL on )
83  {
84  #ifdef GSOFT_FONTS
85  GXT x;
86  GYT y;
87  GXT GFAST w;
88  GYT GFAST h;
89  #endif
90 
91  if (on == 0) glcd_err = 0; /* Reset HW error flag at first call */
92  if( (ghw_cursor&GCURON) == 0 ) /* always kill cursor */
93  on = 0;
94 
95  #ifndef GHW_NO_HDW_FONT
96  if( gishwfont() )
97  {
98  /* Change hardware cursor temporary */
99  GCURSOR c = ghw_cursor;
100  ghw_setcursor( (GCURSOR) (on ? (c | GCURON) : (c & ~GCURON)));
101  ghw_cursor = c;
102  return;
103  }
104  #endif
105 
106  #ifdef GSOFT_FONTS
107  if( on != gi_cursor_on ) /* alter cursor */
108  {
109  gi_cursor_on = on; /* Invert cursor area */
110  if (gcurvp->cpos.x > gcurvp->rb.x)
111  {
112  /* Cursor is temporary out of viewport area (because of GLINECUT or GNOSCROLL) */
113  return;
114  }
115  x = gcurvp->cpos.x;
116  y = gcurvp->cpos.y;
117  w = gcursor_width();
118  h = gcursor_height();
119  if (y < h)
120  y = 0;
121  else
122  y = y - h;
123  if( x+w >= gcurvp->rb.x )
124  w = gcurvp->rb.x-x; /* truncate at VP */
125  if( y+h > gcurvp->rb.y )
126  h = gcurvp->rb.y-y; /* truncate at VP */
127 
128  /* Invert bits under cursor, the rest remain unchanged */
129  ghw_invert(x, y, (GXT)(x+w), (GYT)(y+h));
130  ghw_updatehw();
131  }
132  #endif
133  }

135 /********************************************************************
136  Segment: View port
137  Level: Fonts
138  Cursor is set on/off by a call to gcursorblink()
139  from the application (typically called from a timer function)
140  If SYSFONT the HW support for cursor is used.
141  The cursor is Ored on the graphics currently on the LCD
142 */
143 
144 #ifdef GSOFT_FONTS
145 void gcursorblink(void)
146  {
147  gi_datacheck(); /* check internal data for errors */
148  #ifndef GHW_NO_HDW_FONT
149  if( gishwfont() )
150  return; /* let HW do the job */
151  #endif
152 
153  glcd_err = 0;
154  if( (ghw_cursor & GCURBLINK) == 0 )
155  gi_cursor( 1 ); /* always on */
156  else
157  gi_cursor( (SGBOOL)(!gi_cursor_on) );
158  }
159 #endif /* GSOFT_FONTS */

162 /********************************************************************
163  Segment: Basic text
164  Level: Fonts
165  Sets the cursor type as a combination of GCURSOR
166  Returns the old cursor type
167 */
168 GCURSOR gi_setcursor( GCURSOR type )
169  {
170  GCURSOR c;
171  gi_datacheck(); /* check internal data for errors */
172  gi_cursor( 0 ); /* set cursor off */
173 
174  #ifndef GHW_NO_HDW_FONT
175  if( gishwfont() )
176  {
177  c = ghw_cursor;
178  ghw_setcursor( type );
179  }
180  else
181  #endif
182  {
183  c = ghw_cursor;
184  ghw_setcursor( (GCURSOR)(type & ~GCURON) ); /* set off HW cursor */
185  #ifdef GSOFT_FONTS
186  ghw_cursor = type;
187  gi_cursor((SGBOOL)((type&GCURON) ? 1 : 0)); /* set cursor on/off */
188  #endif
189  }
190 
191  return c;
192  }
193 
194 #endif /* GNOCURSOR */
