49  #ifndef GHW_NO_HDW_FONT

51  return (PGSYMBOL) NULL; /* Illegal font */
52  #endif
53  }

55  cp = (PGCODEPAGE) NULL; /* Use font codepage if any */
56  #ifdef GVIRTUAL_FONTS

58  {
59  PGSYMBOL psym;
60  PGSYMHEADV_V psymh;
61  psym = gi_getsymbol(c, fp, cp);
62  if (psym == NULL)
63  return (PGSYMBOL) NULL;
64 
65  /* Take local copy of virtual symbol header so

67  psymh = psym->vsh.psymh_v;
68  vsymbol_v.numbits = psymh->numbits;
69  vsymbol_v.cxpix = psymh->cxpix;
70  vsymbol_v.cypix = psymh->cypix;

72  vsymbol_v.device_id = gvfdevice(fp);
73
