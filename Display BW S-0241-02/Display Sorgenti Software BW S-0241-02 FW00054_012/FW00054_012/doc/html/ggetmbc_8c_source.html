4 /*
5  Fetch either a character or a multibyte character from string and
6  increment string pointer to the following character (or the terminating 0)
7 
8  When GMULTIBYTE is defined then gi_mbtowc(..) is used internally by
9  the LCD driver library to fetch characters from strings.
10 
11  This module support two different multi-byte implementations
12  1) The "shift value" multibyte mode where a character is 1 or 2 bytes depending
13  on the value of the (first) byte. With G_MULTIBYTE_LIMIT defined as
14  0x80 then the range covered is {0-0x7f} and {0x8000-0xffff}
15  "Shift value mode" is default ( GMULTIBYTE_UTF8 undefined)
16 
17  2) The "UTF-8 mode" where a character is 1,2,3,4,5,6 bytes depending
18  on the value of the first bits in the first byte.
19  If the MSB bit = 1 defines if it is a multibyte value and the
20  following bits in the first byte defines the number of bytes to follow.
21  The UTF-8 mode covers the character value range 0-0x7ffffff
22  "UTF-8 mode" is selected by defining GMULTIBYTE_UTF8 in gdispcfg.h
23 
24  If the LCD driver library should support another multi-byte shift /
25  fetch method then this module is the place to add the modifications.
26 
27  Revision date: 090207
28  Revision Purpose: gi_sizeofmb(..) defined to detect number of byte
29  occupied by (multibyte) character. Isolate multi-byte
30  implementation knowledge to this module.

34  Revision date: 191207

36 
37  Version number: 2.3

39 */

65 GWCHAR gi_mbtowc(PGCSTR *strp, SGUCHAR using_codepage)
66  {
67  GWCHAR wc;
68  /* Fetch first byte */
69  wc = (GWCHAR) *((PGCUCHAR)(*strp)); /* Get byte and convert it to wide char */
70  if ((wc >= G_MULTIBYTE_LIMIT) && /* Is character the first byte of a 2 byte value ?*/
71  (using_codepage != 0)) /* Multibyte characters is only allowed with gputs with codepages */
72  {
73  /* Fetch second byte */
74  (*strp)++;
75  wc = (GWCHAR) (wc * 256) + (GWCHAR) *((PGCUCHAR)(*strp));
76  }

82 /*
83  UTF-8 multibyte mode
84 
85  This implementation is for UTF-8 coded strings.
86  According to the UTF-8 standard a logical character value can occupy 1-6
87  string bytes, covering a logical character value range from 0-0x7fffffff.
88 
89  This implementation only returns values in the range 0x0000-0xffff.
90  Logical values >= 0x10000 are treated as a malformed multibyte and the
91  MALFORMED_RETURNCHAR character value is returned instead.
92  The multi-byte value is however always consumed normally from the string.
93 
94  Oversized multi-bytes (use of multibyte format larger than needed to hold
95  the logical value) are just consumed normally for maximum decoder tolerance
96  and not marked as mal-formed. The only exception is a converted logic value
97  of 0 which will be returned as MALFORMED_RETURNCHAR in order not to confuse
98  a program which use the return value to test for string \0 termination.
99 
100  If a malformed multibyte is detected all string characters up to the next
101  valid character value are consumed and MALFORMED_RETURNCHAR (?) is returned
102  instead, in order to ease debugging.
103 */
104 
105 /* Malformed multibyte.
106  Skip until plain ascii or valid multibyte start or cnt overrun
107 */
108 static SGUCHAR gi_skip_ill_mb(PGCSTR str)
109  {
110  SGUCHAR cnt;
111  for(cnt = 0; cnt < 0xff; cnt++)
112  {
113  if ( (((SGUCHAR)(*str)) < ((SGUCHAR)0x80)) ||
114  ((*str & 0xe0) == 0xc0) ||
115  ((*str & 0xf0) == 0xe0) ||
116  ((*str & 0xf8) == 0xf0) ||
117  ((*str & 0xfc) == 0xf8) ||
118  ((*str & 0xfe) == 0xfc))
119  /* Valid ASCII or start of new multibyte detected */
120  return cnt; /* Number of malformed multibyte values skipped */
121  str++;
122  }
123  return cnt; /* Completely illegal serie of multibyte values */
124  }

131  if (((SGUCHAR)(*str)) < ((SGUCHAR)0x80))
132  return 1;
133  if ((((SGUCHAR)(*str)) & 0xe0) == 0xc0)
134  return 2;

137  if ((((SGUCHAR)(*str)) & 0xf8) == 0xf0)

151 /*
152  Fetch next (multi-byte) character from string.
153  Leave pointer on last char
154 
155  If an illegal multibyte encoding is used a replacement character is returned instead.
156  If using codepage = 0, the return value should be below 255, otherwise the
157  replacement character is returned.
158 */
159 GWCHAR gi_mbtowc(PGCSTR *strp, SGUCHAR using_codepage)
160  {
161  SGUCHAR c,cnt;
162  SGULONG lw;
163  c = (SGUCHAR) *((PGCUCHAR)(*strp)); /* Get byte */
164  if (c < ((SGUCHAR)0x80))
165  return (GWCHAR) c; /* Plain ASCII */
166 
167  /* Multibyte, Do UTF-8 decoding, first bits contains UTF-8 size information */
168  if ((c & 0xe0) == 0xc0)
169  {
170  c &= 0x1f; /* two byte multibyte */
171  cnt = 1;
172  }
173  else
174  if ((c & 0xf0) == 0xe0)
175  {
176  c &= 0x0f; /* three byte multibyte */
177  cnt = 2;
178  }
179  else
180  if ((c & 0xf8) == 0xf0)
181  {
182  c &= 0x07; /* four byte multibyte */
183  cnt = 3;
184  }
185  else
186  if ((c & 0xfc) == 0xf8)
187  {
188  c &= 0x03; /* five byte multibyte */
189  cnt = 4;
190  }
191  else
192  if ((c & 0xfe) == 0xfc)
193  {
194  c &= 0x01; /* six byte multibyte */
195  cnt = 5;
196  }
197  else
198  {
199  /* Malformed multibyte, skip it (them) and return replacement value instead */
200  return (GWCHAR) G_MALFORMED_RETURNCHAR;
201  }
202  /* Fetch second and following bytes */
203  lw = (SGULONG) c;
204  do
205  {
206  /* Get next byte and check it */
207  c = (SGUCHAR) ((PGCUCHAR)(*strp))[1];
208  if ((c & 0xc0) != 0x80)
209  {
210  /* Malformed multibyte, skip it and return replacement value instead */
211  return (GWCHAR) G_MALFORMED_RETURNCHAR;
212  }
213  /* Accepted, was a multibyte continuation char, add it to value */
214  lw = (lw << 6) | ((SGULONG)(c & 0x3f));
215  (*strp)++;
216  }
217  while (--cnt != 0);
218  /* Logical character value assembled */
219 
220  if (lw == 0) /* Do not allow oversized UTF-8 0 to be interpreted as string terminator */
221  return (GWCHAR) G_MALFORMED_RETURNCHAR;
222 
223  /* Check for value oversize */
224  if (using_codepage != 0)
225  {
226  /* For compilers using 32 bit wide char this check can be commented out */
227  if (!(lw <= WCHAR_MAX)) /* > 0xffff for GWCHAR = 2 bytes */
228  /* Library parameter overrun
229  Multibyte value can not be handled by library GWCHAR configuration
230  Return replacement value instead */
231  return (GWCHAR) G_MALFORMED_RETURNCHAR;
232  }
233  else
234  { /* Assume ISO 8859 fonts (256 symbols max) */
235  if (lw >= 256)
236  /* Library font overrun will occur.
237  Multibyte value can not be handled by viewport font type
238  Return replacement value instead */
239  return (GWCHAR) G_MALFORMED_RETURNCHAR;
240  }
241  return (GWCHAR)(lw);
242  }
