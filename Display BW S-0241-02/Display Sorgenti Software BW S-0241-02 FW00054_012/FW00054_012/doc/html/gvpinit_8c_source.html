44 void gi_datacheck(void)
45  {
46  SGUCHAR c;
47  /* Check pointers */
48  if ( gcurvp == NULL )
49  {
50  G_WARNING( "GLCD was not initialized, Initialization will be done" );
51  ginit();
52  }
53  if (gcurvp->pfont == NULL)
54  {
55  G_WARNING( "GLCD was not initialized, Initialization will be done" );
56  ginit();
57  }
58 
59  #ifdef GHW_NO_HDW_FONT
60  if (gishwfont())
61  {
62  G_WARNING( "This display controller does not support hardware fonts" );
63  }
64  #endif
65 
66  /* check VP for illegal values */
67  if( gcurvp->lt.x >= GDISPW ||
68  gcurvp->lt.y >= GDISPH ||
69  gcurvp->rb.x >= GDISPW ||
70  gcurvp->rb.y >= GDISPH ||
71  gcurvp->cpos.x < gcurvp->lt.x ||
72  gcurvp->cpos.y < gcurvp->lt.y ||
73  ((SGUINT) gcurvp->cpos.x > ((SGUINT)gcurvp->rb.x+1)) || /* cpos.x = rb.x+1 is ok, used internally as put stop condition */
74  gcurvp->cpos.y > gcurvp->rb.y ||
75  #ifndef GNOTXTSPACE
76  (gcurvp->rb.x - gcurvp->lt.x) < gcurvp->chln.x ||
77  (gcurvp->rb.y - gcurvp->lt.y) < gcurvp->chln.y ||
78  #endif
79  /*
80  Its legal to have ppos outside view-port and LCD
81  */
82  gcurvp->mode >= GMODELAST )
83  {
84  G_WARNING( "Coordinate range error or missing initialization" );
85  }
86 
87  c = gcurvp->gi_check;
88  gi_calcdatacheck();
89  if( c != gcurvp->gi_check)
90  {
91  c = c; /* for easy set of break point */
92  G_WARNING( "Internal GLCD data error, check sum" );
93  }
94  }

96 /*********************************************************************
97  Segment: Viewport
98  Level: Viewport
99  Calculates the current vp struct check value.
100  Returns the new check value.
101 */
102 void gi_calcdatacheck(void)
103  {
104  SGUCHAR i,s;
105  s = (SGUCHAR)((GVP*)&(gcurvp->gi_check) - gcurvp);
106  gcurvp->gi_check = 0;
107  for( i=0; i<s; i++)
108  {
109  gcurvp->gi_check = gcurvp->gi_check + ((SGUCHAR*)gcurvp)[i];
110  }
111  }

114 void gi_limit_check(void)
115  {
116  GXT w;
117  GYT h;
118  #ifndef GHW_NO_HDW_FONT
119  if( gishwfont() )
120  return;
121  #endif
122  h = gi_fsymh(gcurvp->pfont);
123  w = gi_fsymw(gcurvp->pfont);
124 
125  /* Limit to viewport size */
126  if (w >= (gcurvp->rb.x-gcurvp->lt.x)+1)
127  w = (gcurvp->rb.x-gcurvp->lt.x)+1;
128  if (h >= (gcurvp->rb.y-gcurvp->lt.y)+1)
129  w = (gcurvp->rb.y-gcurvp->lt.y)+1;
130 
131  /* Limit to font size */
132  if (gcurvp->chln.x >= w)
133  {
134  G_WARNING( "Char spacing exceeds font or viewport width, limiting char spacing" );
135  gcurvp->chln.x = w-1;
136  }
137  if (gcurvp->chln.y >= h)
138  {
139  G_WARNING( "Line spacing exceeds font or viewport height, limiting line spacing" );
140  gcurvp->chln.y = h-1;
141  }
142  }

147 /*
148  Internal function clears current viewport
149  Resettype = 0: Clear all
150  Resettype = 1: Clear all, except color setting (gresetvp)
151  Resettype = 2: Clear all, except color and font data setting (gresetposvp)
152 */
153 void gi_resetvp(SGUCHAR resettype)
154  {
155  /* Set full screen coordiantes */
156  gcurvp->lt.x = 0;
157  gcurvp->lt.y = 0;
158  gcurvp->rb.x = GDISPW-1;
159  gcurvp->rb.y = GDISPH-1;
160 
161  #ifdef GGRAPHICS
162  /* reset pixel pos */
163  gcurvp->ppos.x = 0;
164  gcurvp->ppos.y = 0;
165  #endif /* GGRAPHICS */
166 
167  gcurvp->cpos.x = 0;
168  if (resettype <= 1)
169  {
170  /*gcurvp->cpos.y = SYSFONT.symheight-1;*/
171  gcurvp->cpos.y = 8-1; /* default for SYSFONT */
172 
173  /* reset mode */
174  #ifdef GWORDWRAP
175  /* Make compatible with old versions where GWORDWRAP was defined in gdispcfg.h */
176  gcurvp->mode = GWORD_WRAP;
177  #else
178  gcurvp->mode = GNORMAL;
179  #endif
180 
181  /* reset font */
182  gcurvp->pfont = &SYSFONT;
183  /* preset code-page */
184  gcurvp->codepagep = SYSFONT.pcodepage;
185 
186  #ifndef GNOTXTSPACE
187  /* default is no extra character & line spacing */
188  gcurvp->chln.x = 0;
189  gcurvp->chln.y = 0;
190  #endif
191  }
192  else
193  {
194  gcurvp->cpos.y = gi_fsymh(gcurvp->pfont)-1;
195  #ifndef GNOTXTSPACE
196  gi_limit_check(); /* Limit spacing */
197  #endif
198  }
199  gi_calcdatacheck(); /* correct VP checksum to new settings */
200  }

202 /******************************************************************

209 SGUCHAR ginit(void)

218  {

222 
223  /* Reset vp structures */
224  for( i=0; i<GNUMVP; i++ )
225  {
226  gcurvp = &gdata.viewports[i];
227  gi_resetvp(0);
228  }
229 
230  /* select current vp = 0 */
231  gcurvp = &gdata.viewports[0];
232 
233  #ifndef GCONSTTAB
234  /* Clear variable tabulator table */
235  for( i=0; i<GMAXTABS; i++ )
236  gdata.tabs[i] = 0;
237  #endif
238 
239  #ifdef GWIDECHAR
240  gdata.strtype = 0;
241  #endif
242 
243  #ifndef GNOCURSOR
244  gi_cursor_on=0;
245  #endif

248  been detected by the low-level drivers */
