62 typedef enum _GMODE
63  {
64  /* Version <=3 + 4 types */
65  GNORMAL, /* Normal mode (not inversed, not aligned, no partial showing) */
66  GINVERSE = 0x01, /* Inverse (typical white on black) */
67  GNOSCROLL = 0x02, /* Scroll at viewport end is suppressed */
68  GALIGN_LEFT = 0x04, /* String is aligned to the left viewport edge */
69  GALIGN_RIGHT = 0x08, /* String is aligned to the right viewport edge */
70  GALIGN_HCENTER = 0x0c, /* String is centered horizontally */
71  GALIGN_TOP = 0x10, /* String is aligned to the top */
72  GALIGN_BOTTOM = 0x20, /* String is aligned to the bottom */
73  GALIGN_VCENTER = 0x30, /* String is centered vertically */
74 
75  /* Version >= 4 types */
76  GWORD_WRAP = 0x40, /* Use word wrap (default is char wrap) */
77  GNO_WRAP = 0x80, /* Use no character or word wrapping (only \n processing) */
78  GPARTIAL_LINE = 0x100, /* Show lines where symbol height is partly visible */
79  GPARTIAL_CHAR = 0x200, /* Show characters where symbol width can only be partly visible */
80  GVPCLR_UP = 0x0400, /* Clear vp area above text line(s) */
81  GVPCLR_DOWN = 0x0800, /* Clear vp area below text line(s) */
82  GVPCLR_LEFT = 0x1000, /* Clear vp area to the left of text line(s) */ /* default console mode */
83  GVPCLR_RIGHT = 0x2000, /* Clear vp area to the right of text line(s) */
84  GTRANSPERANT = 0x8000, /* Symbol background color is transperant color */
85  GMODELAST = 0xffff
86  } GMODE;
