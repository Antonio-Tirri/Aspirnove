36 void gi_putsymbol(GXT xs,GYT ys,GXT xemax,GYT yemax,
37  PGSYMBOL psymbol, GYT yoffset, SGUINT symsize)
38  {
39  SGUINT xe;
40  SGUINT ye;
41  SGUCHAR bw;
42  SGUCHAR mode; /* hw mode flags */
43  PGSYMBYTE f;
44 
45  #ifdef GVIRTUAL_FONTS
46  if (gissymbolv(psymbol))
47  {
48  #ifdef GDATACHECK
49  /* Check for mixed font design error */
50  if (((((PGSYMHEADV)psymbol)->type_id) & GVTYPEMASK) != 1)
51  { /* Error: format is unknown to this library */
52  G_WARNING( "Unsupported extended font format. Illegal virtual font" );
53  return;
54  }
55  if ((((PGSYMHEADV)psymbol)->psymh_v->numbits) != 0)
56  { /* Error Color symbols are not supported by b&w libs */
57  G_WARNING( "Color symbols are not supported by b&w libs" );
58  return;
59  }
60  #endif
61  f = (PGSYMBYTE) NULL; /* Signal use of virtual font to low-level driver*/
62  }
63  else
64  #endif
65  f = (PGSYMBYTE)&(((PGBWSYMBOL)psymbol)->b[0]); /* first byte in symbol */
66  mode = G_IS_INVERSE() ? GHW_INVERSE : 0;
67 
68  xe = (SGUINT) gsymw(psymbol);
69  ye = (SGUINT) gsymh(psymbol);
70  if ((ye == 0) || (xe == 0))
71  {
72  /* Error: illegal symbol header data. Corrupted data or a ROM/RAM pointer problem detected */
73  G_WARNING( "Illegal symbol header data detected" );
74  return;
75  }
76 
77  if (symsize == 0)
78  bw = (SGUCHAR) ((xe+7)/8); /* width of symbol in bytes */
79  else
80  bw = (SGUCHAR) (symsize/ye);
81 
82  if (G_IS_TRANSPERANT())
83  mode |= GHW_TRANSPERANT;
84 
85  if (yoffset != 0)
86  { /* Only showing lower part of symbol */
87  ys = ys + yoffset;
88  ye = ye - yoffset;
89  #ifdef GVIRTUAL_FONTS
90  if (!gissymbolv(psymbol))
91  #endif
92  f=&f[(SGUINT)bw*yoffset];
93  }
94 
95  xe = xe+((SGUINT) xs)-1;
96  ye = ye+((SGUINT) ys)-1;
97 
98  /* truncate at max rect */
99  if( xe > ((SGUINT) xemax) )
100  xe = (SGUINT) xemax;
101  if( ye > ((SGUINT) yemax) )
102  ye = (SGUINT) yemax;
103 
104  glcd_err = 0; /* Reset HW error flag */
105 
106  #ifdef GVIRTUAL_FONTS
107  if (gissymbolv(psymbol))
108  gi_symv_open( psymbol, bw, yoffset ); /* Preset virtual symbol interface */
109  #endif
110 
111  ghw_wrsym(xs, ys, (GXT) xe, (GYT) ye, f, bw, mode );
112  }
